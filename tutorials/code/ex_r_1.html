make a quick time series plot using the default settings

## graphics::plot.default loads graphics package and runs plot with default settings
graphics::plot.default(
  x = economics$date,
  y = economics$pce,
  type = "l",
  xlab = "Date",
  ylab = "index=1000",
  main = "Personal Consumption Expenditures (graphics::plot.default)",
  col  = "green"
)
